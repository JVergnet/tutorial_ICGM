# contour plot
n_points = 20
aa = np.linspace(-7, 7, n_points)
bb = np.linspace(-3, 3, n_points)

def cost_1(a, b):
    return np.sin(2*np.sqrt((a**2+b**2))) # get inventive here ! ;-) 

z_1 = []
for a in aa:
    for b in bb:
        z_1.append(cost_1(a, b))
z_1 = np.reshape(z_1, [len(aa), len(bb)])

plt.contourf(aa, bb, z_1)

plt.colorbar()
